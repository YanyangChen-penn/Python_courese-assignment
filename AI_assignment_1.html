POLYS = r"data/data/philadelphia-census-tract.shp"

polygons = []
with fiona.open(POLYS, "r") as poly_src:
    poly_crs = None
    # 优先用 wkt，没有再用 crs dict
    if getattr(poly_src, "crs_wkt", None):
        poly_crs = CRS.from_wkt(poly_src.crs_wkt)
    elif poly_src.crs:
        poly_crs = CRS.from_user_input(poly_src.crs)

    poly_schema = poly_src.schema.copy()   # 原 schema 备用（写出时用）

    for feat in poly_src:
        geom = shape(feat["geometry"])     # Shapely Polygon/MultiPolygon
        polygons.append({"geom": geom, "props": dict(feat["properties"])})

# Rtree：用面要素的包围盒建索引
rt = index.Index()
for i, p in enumerate(polygons):
    rt.insert(i, p["geom"].bounds)

# 预编译 polygon（提高 contains/within 判断效率）
prepped_polys = [prep(p["geom"]) for p in polygons]

POINTS = r"data/data/crash_data_collision_crash_2007_2017.shp"

# 想筛的条件（演示 & |）：
year = 2015
months = [5, 6]       # 5月 或 6月   —— 使用“或”（OR）
weekdays = [1, 5]     # 周一 或 周五 —— 使用“或”（OR）

points = []
with fiona.open(POINTS, "r") as pt_src:
    # 处理 CRS（和普查区一致时无需转换）
    pt_crs = None
    if getattr(pt_src, "crs_wkt", None):
        pt_crs = CRS.from_wkt(pt_src.crs_wkt)
    elif pt_src.crs:
        pt_crs = CRS.from_user_input(pt_src.crs)

    need_transform = (poly_crs is not None and pt_crs is not None and poly_crs != pt_crs)
    transformer = Transformer.from_crs(pt_crs, poly_crs, always_xy=True) if need_transform else None

    n_total = 0
    n_kept = 0

    for feat in pt_src:
        n_total += 1
        props = feat["properties"]
        y_ok = (props.get("crash_year") == year)                 # AND
        m_ok = (props.get("crash_mont") in months)               # (5 或 6)
        d_ok = (props.get("day_of_wee") in weekdays)             # (周一 或 周五)

        if y_ok and (m_ok) and (d_ok):                           # AND 的组合
            geom = shape(feat["geometry"])
            if need_transform:
                geom = transform(lambda x, y, z=None: transformer.transform(x, y), geom)
            points.append({"geom": geom, "props": props})
            n_kept += 1

print(f"读入点要素：{n_total} 条；符合条件：{n_kept} 条")

# 计数与数值统计容器
counts = Counter()                      # 每个面里有多少点
vals_mean_median = defaultdict(list)    # 用于存放每个面的数值列表（例如 max_severi）

severity_field = "max_severi"           # 若没有此字段，可换其它数值字段名或置 None

for pt in points:
    g = pt["geom"]
    # 先用 Rtree 粗筛候选多边形
    candidates = list(rt.intersection(g.bounds))
    # 精确判断：点是否在面内
    for pid in candidates:
        if prepped_polys[pid].contains(g):      # 对点来说 contains/within 等价
            counts[pid] += 1
            if severity_field is not None:
                v = pt["props"].get(severity_field)
                if v is not None:
                    try:
                        vals_mean_median[pid].append(float(v))
                    except:
                        pass
            break  # 一个点只记入一个面

# 1) 准备着色：这次用灰度色图
counts = [p["props"].get("acc_ct", 0) for p in polygons]
vmin, vmax = 0, max(counts) if counts else 1
norm = Normalize(vmin=vmin, vmax=vmax)

cmap = matplotlib.colormaps["Greys"]  # 灰度底图
sm = ScalarMappable(norm=norm, cmap=cmap)

# 2) 设置黑底
fig, ax = plt.subplots(figsize=(12, 10), facecolor="black")
ax.set_facecolor("black")

# tract 多边形（灰度表示事故数量）
for p in polygons:
    geom = p["geom"]
    c = p["props"].get("acc_ct", 0)
    facecolor = cmap(norm(c))

    def draw_poly(poly):
        x, y = poly.exterior.xy
        ax.fill(x, y, facecolor=facecolor, edgecolor="dimgray", linewidth=0.3)
        for interior in poly.interiors:
            xi, yi = interior.xy
            ax.plot(xi, yi, color="dimgray", linewidth=0.3)

    if isinstance(geom, Polygon):
        draw_poly(geom)
    elif isinstance(geom, MultiPolygon):
        for sub in geom.geoms:
            draw_poly(sub)
